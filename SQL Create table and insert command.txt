
drop table arrears, expenses,savings,earnings,fine,maintenance,months,members;

create table members(flat_no int unsigned not null primary key,name varchar(25) not null, mobile_number varchar(12) not null);

create table months(month_id varchar(15) not null primary key, month_name varchar(20) not null, previous_month_id varchar(15) not null, created_date datetime not null,latest tinyint(1) not null);

create table maintenance(maintenance_id int unsigned not null auto_increment primary key, maintenance_flat_no int unsigned not null,maintenance_month_id varchar(15) not null,maintenance_status tinyint(1) unsigned not null, maintenance_amount int unsigned not null, maintenance_date datetime not null, foreign key(maintenance_flat_no) references members(flat_no), foreign key(maintenance_month_id) references months(month_id));

create table fine(fine_id int unsigned not null auto_increment primary key, fine_flat_no int unsigned not null,fine_month_id varchar(15) not null,fine_status tinyint(1) unsigned not null, fine_amount int unsigned not null, fine_date datetime not null, foreign key(fine_flat_no) references members(flat_no), foreign key(fine_month_id) references months(month_id));

create table earnings(earning_id int unsigned not null auto_increment primary key, earning_month varchar(15) not null, earning_title text not null, earning_amount int unsigned not null,earning_date datetime not null, earning_photo varchar(50) not null, foreign key(earning_month) references months(month_id));

create table expenses(expense_id int unsigned not null auto_increment primary key, expense_month varchar(15) not null, expense_title text not null, expense_amount int unsigned not null,expense_date datetime not null, expense_photo varchar(50) not null, foreign key(expense_month) references months(month_id));

create table savings(saving_month_id varchar(15) not null, maintenance_total int not null, fine_total int not null, previous_month_total int not null,expenses_total int not null, earnings_total int not null, total int not null, note text,foreign key(saving_month_id) references months(month_id));

create table arrears(arrears_amount int not null, previous_arrears int not null, arrears_month_id varchar(15) not null, foreign key(arrears_month_id) references months(month_id));

create table visitors(visitor_id varchar(21) not null primary key,ip varchar(40),visit_count int not null,last_visit datetime not null);
